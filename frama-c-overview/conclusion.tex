\begin{frame}{Conclusion}

  We have presented how to:
  \smallskip
  \begin{itemize}
  \item verify the absence of runtime errors with \blue{Eva}
    \smallskip
  \item formally specify functional properties with \blue{ACSL}
    \smallskip
  \item prove a programs respects its specification with \blue{WP}
    \smallskip
  \item verify annotations at runtime or detect runtime errors with \blue{E-ACSL}
  \end{itemize}

  \begin{center}
    {\large \red{All of these and much more inside Frama-C}}
    \medskip
  \end{center}

  May be used for:
  \begin{itemize}
  \smallskip
  \item \blue{teaching}
  \smallskip
  \item \blue{academic} prototyping
  \smallskip
  \item \blue{industrial} applications
  \end{itemize}

  \begin{center}
    {\large \red{http://frama-c.com}}
  \end{center}

\end{frame}
